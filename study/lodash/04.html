# Lodash 函数常用方法的详细使用示例

# 1、_.debounce：防抖函数，用于延迟执行函数。例如：

function search() {
  console.log('searching...');
}

const debounceSearch = _.debounce(search, 1000);
debounceSearch(); // 1s 后输出 'searching...'

# 2、_.throttle：节流函数，用于控制函数执行频率。例如：

function scroll() {
  console.log('scrolling...');
}

const throttleScroll = _.throttle(scroll, 1000);
window.addEventListener('scroll', throttleScroll); // 每 1s 最多执行一次

# 3、_.once：只执行一次的函数。例如：

function init() {
  console.log('initializing...');
}

const initOnce = _.once(init);
initOnce(); // 输出 'initializing...'
initOnce(); // 不再输出

# 4、_.delay：延迟执行函数。例如：

function sayHi(name) {
  console.log(`Hi, ${name}!`);
}

_.delay(sayHi, 1000, 'Alice'); // 1s 后输出 'Hi, Alice!'

# 5、_.flow：组合多个函数为一个函数，从左到右依次执行。例如：

function double(n) {
  return n * 2;
}

function addOne(n) {
  return n + 1;
}

const doubleThenAddOne = _.flow([double, addOne]);
console.log(doubleThenAddOne(3)); // 输出 7（3 * 2 + 1）

# 6、_.memoize：记忆函数，缓存函数执行结果，避免重复计算。例如：

function expensiveCalculation(n) {
  console.log(`calculating ${n}...`);
  return n * 2;
}

const memoizedCalculation = _.memoize(expensiveCalculation);
console.log(memoizedCalculation(3)); // 输出 'calculating 3...' 和 6
console.log(memoizedCalculation(3)); // 只输出 6，不再计算
