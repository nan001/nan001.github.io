# es5 严格模式

ECMAScript 5.1标准中引入了严格模式（Strict Mode），它可以让JavaScript在更严格的条件下运行，并且消除了一些不安全、不规范的语法和行为。严格模式可以通过在脚本或函数开头添加'use strict';来启用。以下是严格模式的详细解释和示例：

# 严格模式的作用

消除了JavaScript语法的一些不合理、不严谨之处，减少了一些怪异行为；
消除代码运行的一些不安全之处，保证代码运行的安全；
提高编译器效率，增加运行速度；
为未来新版本的JavaScript做好铺垫。

# 严格模式的语法

在全局作用域或函数内部的第一条语句之前，添加字符串'use strict';即可启用严格模式。

'use strict';

function test() {
  'use strict';
  // 函数内部启用严格模式
}

# 严格模式的变化

启用严格模式后，JavaScript语言的一些行为将发生变化，以下是主要的变化：

# 变量必须先声明后使用，否则将报错。

'use strict';

x = 1;  // 报错

# 函数中的this关键字指向undefined，而不是全局对象。

'use strict';

function test() {
  console.log(this);  // 输出undefined
}

test();

# 对象不能有重名的属性。

'use strict';

var obj = {
  prop1: 'value1',
  prop1: 'value2'  // 报错
};

# 禁止使用with语句。

'use strict';

var obj = {
  prop1: 'value1'
};

with (obj) {  // 报错
  console.log(prop1);
}

# 禁止使用eval和arguments作为变量名。

'use strict';

var eval = 1;  // 报错

function test(arguments) {  // 报错
  console.log(arguments);
}

# 不能删除变量或对象的属性。

'use strict';

var x = 1;

delete x;  // 报错

var obj = {
  prop1: 'value1'
};

delete obj.prop1;  // 报错

# 严格模式的示例

以下是一些严格模式的示例代码：

# 禁止使用未声明的变量。

'use strict';

function test() {
  x = 1;  // 报错
}

test();

# 函数中的this关键字指向undefined。

'use strict';

function test() {
  console.log(this);  // 输出undefined
}

test();

# 禁止使用with语句。

'use strict';

var obj = {
prop1: 'value1'
};

// 禁止使用with语句
with (obj) { // 报错
console.log(prop1);
}

# 不能删除变量或对象的属性。

'use strict';

var x = 1;

delete x;  // 报错

var obj = {
  prop1: 'value1'
};

delete obj.prop1;  // 报错

# 函数的参数不能有重名的。

'use strict';

function test(x, x) {  // 报错
  console.log(x);
}

test(1, 2);

# 禁止使用eval和arguments作为变量名。

'use strict';

var eval = 1;  // 报错

function test(arguments) {  // 报错
  console.log(arguments);
}

# 对象不能有重名的属性。

'use strict';

var obj = {
  prop1: 'value1',
  prop1: 'value2'  // 报错
};

# eval函数和arguments对象的行为发生了变化。

'use strict';

var x = 1;

function test() {
  var x = 2;

  console.log(eval('x'));      // 输出1
  console.log(arguments[0]);  // 输出2
}

test(2);

以上是严格模式的详细解释和示例，通过使用严格模式，可以让JavaScript在更严格、更安全的环境下运行，并且能够更好地遵守JavaScript的规范。
