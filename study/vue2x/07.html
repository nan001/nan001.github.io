# vue项目--Axios请求拦截


import Vue from 'vue'
import axios from 'axios'
import qs from 'qs'
import codeMessage from './plugins/codeStatus.js'
import {
	Message,
	Loading,
} from 'element-ui'
Vue.use(Loading)
Vue.prototype.$message = Message;

/* request 拦截器 start */
var loadingCount = 0;
let loadingInstance
axios.interceptors.request.use(
    config => {
        loadingInstance = Loading.service({ // 全局Loading
            text: '玩命加载中...',
            spinner: 'el-icon-loading',
            background: 'rgba(255, 255, 255, 0.8)'
        });
        //判断是否存在token
        if (store.state.login.Authorization) {
            config.headers.common['Authorization'] = store.state.login.Authorization
        }
        loadingCount++;
        if (config.method === 'post' && config.data.constructor !== FormData) {
            config.data = qs.stringify(config.data)
        }
        return config
    }, error => {
        loadingInstance.close()
        loadingCount--;
        if (loadingCount === 0) {
            // return
        }
        Vue.prototype.$message({
            showClose: true,
            message: '加载超时',
            type: 'error'
        })
        return Promise.reject(error)
    }
);
/* request 拦截器 end */

/* response 拦截器 start */
axios.interceptors.response.use(
    response => {
        loadingInstance.close() // 关闭全局Loading
		
        if (response.headers.authorization) {
            store.dispatch('login/setAuthorization', response.headers.authorization)
        }
        if (response.headers.tokenexpired == 'expired') {
            Vue.prototype.$message({
                showClose: true,
                message: '登录状态失效，请先登录！',
                type: 'error'
            })
            store.dispatch('login/del_Authorization')
            router.push({
                path: '/login'
            })
        }
		
        return response;
    },
    error => {
        loadingInstance.close()
        if (error.response) {
            if (error.response.headers.authorization) {
                store.dispatch('login/setAuthorization', error.response.headers.authorization)
            }
			Vue.prototype.$message({
			    showClose: true,
			    message: codeMessage[response.data.status],
			    type: 'error'
			})
			if(error.response.status == 401){
				store.dispatch('login/del_Authorization')
				router.push({
				    path: '/login'
				})
			}
        }
        let {message} = error;
        if (message == 'Network Error')
            return Promise.reject(error)
        return Promise.reject(error.response.data)
    }
);
/* response 拦截器 end */

const codeMessage = {
	200: '服务器成功返回请求的数据。',
    201: '新建或修改数据成功。',
    202: '一个请求已经进入后台排队(异步任务)。',
    204: '删除数据成功。',
    304: '已经执行了GET,但文件未变化',
    400: '发出的请求有错误,服务器没有进行新建或修改数据的操作。',
    401: '用户没有权限,令牌、用户名、密码错误。',
    402: '参数失败',
    403: '用户得到授权,但是访问是被禁止的。',
    404: '发出的请求针对的是不存在的记录,服务器没有进行操作。',
    406: '请求的格式不可得。',
    410: '请求的资源被永久删除,且不会再得到的。',
    422: '当创建一个对象时,发生一个验证错误。',
    500: '服务器发生错误,请检查服务器。',
    502: '网关错误。',
    503: '服务不可用,服务器暂时过载或维护。',
    504: '网关超时',
}
