# vue项目--Axios配置不同的baseURL,请求不同的域名接口

# 一、在plugins文件夹中，新建config.js、https.js

  • config.js 内容

//1->config.js中引入axios
import Axios from 'axios'
Axios.defaults.timeout = 10000
Axios.defaults.headers.post['Content-Type'] = 'application/x-www-form-urlencoded';


//2->对get请求传递过来的参数处理
function paramsToUrl(url, params) {
	if (!params) return url
	for (var key in params) {
		if (params[key] && params[key] != 'undefined') {
			if (url.indexOf('?') != -1) {
				url += '&' + '' + key + '=' + params[key] || '' + ''
			} else {
				url += '?' + '' + key + '=' + params[key] || '' + ''
			}
		}
	}
	return url
}

//3->在config.js中封装公用方法-----封装请求的方法
function requireData(url, params, type, item) {
	if (!url) return false
	switch (item) {
		case 'M1':
			url = Axios.defaults.baseM1URL + url
			break;
		case 'M2':
			url = Axios.defaults.baseM2URL + url
			break;
		default:
			url = Axios.defaults.baseM1URL + url
	}
	if (type === 'get') {
		url = paramsToUrl(url, params)
		return new Promise((resolve, reject) => {
			Axios.get(url, params).then(res => {
				resolve(res)
			}).catch(err => {
				reject(err)
			})
		})
	} else {
		return new Promise((resolve, reject) => {
			Axios.post(url, params).then(res => {
				resolve(res)
			}).catch(err => {
				reject(err)
			})
		})
	}
}

export default {
	install(Vue, options) {
		//->接口1的请求数据方法
		Vue.prototype.$wxapi = function(url, params, type) {
			return requireData.call(this, url, params, type, 'M1')
		}
		//->接口2的请求数据方法
		Vue.prototype.$apkapi = function(url, params, type) {
			return requireData.call(this, url, params, type, 'M2')
		}
	}
}
  • https.js 内容

// 导出 对应的不同接口的URL地址
module.exports = {
	domain: {
		Base_M1_URL: 'https://xxx.xxx.com/wxapi', //模块一接口地址
		Base_M2_URL: 'https://xxx.xxx.com/apkapi', //模块二接口地址
	}
}

# 二、在main.js中使用


// 配置http访问 start
import {
	domain
} from './plugins/https.js'
import common from './plugins/config'
Vue.use(common)
axios.defaults.baseM1URL = domain.Base_M1_URL
axios.defaults.baseM2URL = domain.Base_M2_URL
// 配置http访问 end
