# vue项目--配置打包压缩

  • 配置压缩 CDN px转vw

const path = require('path')
const webpack = require('webpack')
const autoprefixer = require('autoprefixer');
const pxtoviewport = require('postcss-px-to-viewport');

const CompressionWebpackPlugin = require('compression-webpack-plugin') // 压缩文件
const productionGzipExtensions = ['js', 'css']
const isProduction = process.env.NODE_ENV === 'production'

module.exports = {
	// 压缩文件 start
	lintOnSave: false,
	runtimeCompiler: true,
	publicPath: './', // 设置打包文件相对路径
	// 这是前端解决跨域的代码
	devServer: {
		// open: process.platform === 'darwin',
		// host: 'localhost',
		port: 8080,
		// open: true, //配置自动启动浏览器
		proxy: {
			'/api': {
				target: ``, //写上接口基地址
				changeOrigin: true,
				ws: true,
				// secure: false, //如果是http接口，需要配置此参数
				pathRewrite: {
					'^/api': ''
				}
			}
		}
	}, 
	configureWebpack: {
		resolve: {
			alias: {
				'@': path.resolve(__dirname, './src'),
				'@i': path.resolve(__dirname, './src/assets'),
			}
		},
		plugins: [
			new webpack.IgnorePlugin(/^\.\/locale$/, /moment$/),

			// 下面是下载的插件的配置
			new CompressionWebpackPlugin({
				algorithm: 'gzip',
				test: new RegExp('\\.(' + productionGzipExtensions.join('|') + ')$'),
				threshold: 10240,
				minRatio: 0.8
			}),
			// 合并小文件
			new webpack.optimize.LimitChunkCountPlugin({
				maxChunks: 20,  // 最多合并文件数量（及最终文件数量）
				minChunkSize: 100 // 要合并的文件最小大小
			})
		]
	},
	// 压缩文件 end
	/* 配置  cdn  start */
	chainWebpack: config => {
		const types = ['style-loader', 'css-loader', 'less-loader']
		types.forEach(type => addStyleResource(config.module.rule('less').oneOf(type)))
		// 查看打包文件体积大小
		config.plugin('webpack-bundle-analyzer').use(require('webpack-bundle-analyzer').BundleAnalyzerPlugin)
		if (process.env.NODE_ENV === 'production') {
			var externals = {
				// vue: 'Vue',
			}
			config.externals(externals)
			const cdn = {
				css: [
					// Swiper
					// 'https://cdn.bootcdn.net/ajax/libs/Swiper/3.4.2/css/swiper.min.css'
				],
				js: [
					// vue 
					// 'https://cdn.bootcdn.net/ajax/libs/vue/2.6.11/vue.min.js',
				]
			}
			// 通过 html-webpack-plugin 将 cdn 注入到 index.html 之中
			config.plugin('html')
				.tap(args => {
					args[0].cdn = cdn
					return args
				})
		}
	},
	/* 配置  cdn  end */
	css: {
		extract: false,
		loaderOptions: {
			postcss: {
				plugins: [
					autoprefixer(),
					pxtoviewport({
						viewportWidth: 375
					})
				]
			},
			less: {
				lessOptions: {
					modifyVars: {
						hack: 'true; @import "/src/assets/style/var.less";'
					}
				},
			},
		},
	},
	productionSourceMap: false,
	outputDir: 'h5_name',
	assetsDir: 'static'
};

function addStyleResource(rule) {
	rule.use('style-resource')
		.loader('style-resources-loader')
		.options({
			patterns: [
				path.resolve(__dirname, '/src/assets/style/var.less'),
			],
		})
}
